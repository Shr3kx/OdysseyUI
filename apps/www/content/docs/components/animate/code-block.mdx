<ComponentInstallation name="components-animate-code-block" />

## Usage

```tsx
<CodeFrame
  code={`console.log("Hello from JavaScript");`}
  language="javascript"
/>
```

## API Reference

### CodeFrame

<TypeTable
  type={{
    code: {
      description: 'The code string to be copied.',
      type: 'string',
      required: true,
    },
    language: {
      description: 'The programming language displayed in the badge.',
      type: 'string',
      required: false,
      default: 'JS',
    },
    className: {
      description: 'Additional CSS classes for the frame container.',
      type: 'string',
      required: false,
    },
    children: {
      description: 'The rendered code content to display inside the frame.',
      type: 'React.ReactNode',
      required: true,
    },
  }}
/>

## Examples

### With Different Languages

```tsx
  <CodeFrame
  code={`console.log("Hello from JavaScript");`}
  language="javascript"
/>

<CodeFrame
  code={`print("Hello from Python")`}
  language="python"
/>

<CodeFrame
  code={`System.out.println("Hello from Java");`}
  language="java"
/>
```

### With Custom Styling

```tsx
<CodeFrame
  code={`console.log("Styled CodeFrame");`}
  language="javascript"
  className="rounded-xl border border-neutral-200 bg-neutral-50"
/>
```
