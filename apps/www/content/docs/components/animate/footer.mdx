## Usage

```tsx
import Footer from '@/components/odysseyui/footer';

export default function Page() {
  return (
    <Footer
      theme="dia-browser"
      copyrightText={`copyright © ${new Date().getFullYear()} — Odyssey UI`}
    />
  );
}
```

Available `theme` presets: `dia-browser`, `ocean`, `amber`, `emerald`, `violet`, `rose`.
